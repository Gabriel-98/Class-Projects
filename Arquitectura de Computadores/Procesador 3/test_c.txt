int multiplicacion(int a, int b){
	int k=a,c=b,x,y,ans=0;
	while(c>0){
		x = c >> 1;
		y = x << 1;
		if(c-y == 1)
		ans += k;
		c = x;
		k <<= 1;
	}
	return ans;
}

int main(){
	int a,b,ans;
	a=23;
	b=75;
	ans= multiplicacion(a,b);
}
